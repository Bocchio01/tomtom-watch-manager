87 {
88 TxPacket tx_copy = packet;
89 tx_copy.header.counter = current_counter_++;
90
91 // Adjust length field to include the extra data
92 // Header length = (Counter + Type + StructuredPayload) + ExtraData
93 tx_copy.header.length += static_cast<uint8_t>(extra_data.size());
94
95 std::vector<uint8_t> buffer;
96 buffer.reserve(tx_copy.size() + extra_data.size());
97
98 // Serialize structured packet
99 const uint8_t *pkg_ptr = reinterpret_cast<const uint8_t *>(&tx_copy);
100 buffer.insert(buffer.end(), pkg_ptr, pkg_ptr + tx_copy.size());
101
102 // Append extra data
103 if (!extra_data.empty())
104 {
105 buffer.insert(buffer.end(), extra_data.begin(), extra_data.end());
106 }
107

109 if (result < 0)

121 template <typename RxPacket>
122 PacketResponse<RxPacket> receive()
123 {
124 PacketResponse<RxPacket> response;
125
126 // Read the header first
127 std::vector<uint8_t> header_bytes(sizeof(definitions::PacketHeader));
128 connection_->read(header_bytes.data(), header_bytes.size(), 2000);
129
130 // Read the payload based on length in header
131 std::vector<uint8_t> payload_bytes(header_bytes[1] - 2); // Subtract counter(1) + type(1)
132 connection_->read(payload_bytes.data(), payload_bytes.size(), 2000);
133
134 // Construct the packet header
135 std::memcpy(&response.packet.header, header_bytes.data(), header_bytes.size());
136
137 // Copy structured payload (only up to sizeof(payload))
138 size_t copy_size = std::min(payload_bytes.size(), sizeof(response.packet.payload));
139 std::memcpy(&response.packet.payload, payload_bytes.data(), copy_size);
140
141 // Store raw payload bytes for variable-length access
142 response.raw_payload_bytes = std::move(payload_bytes);
143
144 // Clear any remaining data in the connection buffer
145 connection_->read(nullptr, 0, 0);
146
147 return response;
148 }
